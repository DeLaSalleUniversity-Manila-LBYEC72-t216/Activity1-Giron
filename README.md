#include <stdio.h>
#include <stdlib.h>



int main(int argc, char *argv[]) {
	
	int area, base, height;
	
	
	printf("Input triangle's base: ");
	scanf("%d", &base);
	
	printf("Input triangle's height: ");
	scanf("%d", &height);
	
	area=(base*height)/2;
	
	printf("The triangle's area is %d", area);
	return 0;
}
